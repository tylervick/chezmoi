#!/bin/zsh
#
# .zsh_plugins.txt - antidote plugins file
#
# antidote generates a static .zsh_plugins.zsh from this file, which .zshrc sources.
# Order matters: prerequisites first, prompt mid, syntax-highlighting/history-search last.
#

### Completions
#
# Completions must be added to fpath BEFORE compinit runs. ez-compinit handles
# compinit for us, so completion plugins (kind:fpath) go above it.
###

# pnpm completions
empresslabs/pnpm.plugin.zsh path:completions kind:fpath

# Supplemental completions for many common tools.
zsh-users/zsh-completions path:src kind:fpath

# Handles compinit + completion zstyles so we don't have to.
mattmc3/ez-compinit


### Regular Plugins
###
wintermi/zsh-brew


### Local Plugins (conf.d/ snippets + aliases + prompt config)
#
# These are sourced in order. mise/zoxide are activated here via `eval` rather than
# pulling the oh-my-zsh plugins (dropped 2026 — see git history / plan).
###
$ZDOTDIR/.aliases
$ZDOTDIR/.p10k.zsh
$ZDOTDIR/conf.d/brew.zsh
$ZDOTDIR/conf.d/history.zsh
$ZDOTDIR/conf.d/mise.zsh
$ZDOTDIR/conf.d/zoxide.zsh
$ZDOTDIR/conf.d/jira.zsh
$ZDOTDIR/conf.d/rustup.zsh
$ZDOTDIR/conf.d/colima.zsh


### Prompt
#
# powerlevel10k. STARSHIP-TODO: a future migration to starship is planned (see plan /
# git history). When switching, remove this bundle, brew "powerlevel10k", the p10k
# instant-prompt block in .zshrc, and the $ZDOTDIR/.p10k.zsh local plugin above.
###
romkatv/powerlevel10k


### Deferred Plugins
#
# Deferred via romkatv/zsh-defer — loaded once zsh is idle, speeding up startup.
###

# Syntax highlighting
zdharma-continuum/fast-syntax-highlighting kind:defer


### Final Plugins
#
# Order matters at the end: autosuggestions before history-substring-search.
###
mattmc3/zfunctions
zsh-users/zsh-autosuggestions kind:defer
zsh-users/zsh-history-substring-search
